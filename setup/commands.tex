% For some papers, we define new notation that only works in math mode. Ensure
% that these can be easily mentioned in text, with the proper spacing, using the
% following wrapper command.
\newcommand{\emxs}[1]{\ensuremath{#1}\xspace}

% Improve formatting of paragraphs.
\renewcommand{\paragraph}[2][.]{\smallskip\noindent\textbf{#2#1}}

% Required: enumitem (with inline option)
% A more compact enumerate style (smaller left margins and less spacing between
% items).
\newenvironment{compactenum}[1][\arabic*.]
{\begin{enumerate}[label={#1}, noitemsep]}
{\end{enumerate}}
% A more compact itemize style.
\newenvironment{compactitem}
{\begin{itemize}[noitemsep]}
{\end{itemize}}
% Enumerated lists inside paragraphs.
\newenvironment{inparaenum}[1][(\arabic*)]
{\begin{enumerate*}[label={#1}, itemjoin={{, }}, itemjoin*={{, and }}]}
{\end{enumerate*}}
